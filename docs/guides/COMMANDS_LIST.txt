════════════════════════════════════════════════════════════════
📂 COMPUTER CONTROL AGENT
════════════════════════════════════════════════════════════════

▶ agent.execute()                    - تنفيذ عملية واحدة
▶ agent.executeTask()                - تنفيذ مهمة كاملة
▶ agent.pause()                      - إيقاف مؤقت
▶ agent.resume()                     - استئناف
▶ agent.stop()                       - إيقاف كامل
▶ agent.emergencyStop()              - إيقاف طارئ
▶ agent.isAgentActive()              - فحص إذا كان نشط
▶ agent.isAgentPaused()              - فحص إذا كان متوقف
▶ agent.getSystemState()             - حالة النظام
▶ agent.getOperationHistory()        - سجل العمليات
▶ agent.getOperation()               - معلومات عملية محددة
▶ agent.exportLogs()                 - تصدير السجلات
▶ agent.createSnapshot()             - إنشاء snapshot
▶ agent.restoreFromSnapshot()        - استعادة snapshot
▶ agent.listSnapshots()              - قائمة snapshots
▶ agent.enableDeadManSwitch()        - تفعيل Dead Man Switch
▶ agent.disableDeadManSwitch()       - إيقاف Dead Man Switch
▶ agent.heartbeat()                  - إعادة ضبط Dead Man Switch
▶ agent.updateSecurityPolicy()       - تحديث سياسة الأمان
▶ agent.getStatus()                  - حالة الوكيل
▶ agent.getRecentErrors()            - الأخطاء الأخيرة

🏷️ أنواع العمليات (Operation Types)
════════════════════════════════════════════════════════════════

type: 'file-read'              - قراءة ملف

type: 'file-copy'              - نسخ ملف
type: 'dir-create'             - إنشاء مجلد
type: 'dir-delete'             - حذف مجلد
type: 'process-start'          - بدء عملية
type: 'process-kill'           - إنهاء عملية
type: 'network-request'        - طلب شبكة
type: 'system-command'         - أمر نظام
type: 'package-install'        - تثبيت حزمة
type: 'service-manage'         - إدارة خدمة
type: 'database-query'         - استعلام قاعدة بيانات
type: 'git-operation'          - عملية Git
type: 'custom'                 - عملية مخصصة

════════════════════════════════════════════════════════════════
🔒 مستويات الأمان (Operation Levels)
════════════════════════════════════════════════════════════════

level: 'safe'                  - آمن (قراءة فقط)
level: 'medium'                - متوسط (كتابة/تعديل)
level: 'critical'              - حرج (حذف/تنفيذ)

════════════════════════════════════════════════════════════════
🛑 مستويات الإيقاف الطارئ (Emergency Stop Levels)
════════════════════════════════════════════════════════════════

'pause'                        - إيقاف مؤقت
'soft'                         - إيقاف ناعم
'hard'                         - إيقاف قوي
'emergency'                    - طوارئ + rollback
'panic'                        - ذعر (إيقاف فوري)

════════════════════════════════════════════════════════════════
📢 الأحداث (Events)
════════════════════════════════════════════════════════════════

agent.on('operation:start')    - عند بدء عملية
agent.on('operation:complete') - عند إكمال عملية
agent.on('operation:error')    - عند خطأ في عملية
agent.on('emergency:stop')     - عند إيقاف طارئ
agent.on('snapshot:created')   - عند إنشاء snapshot
monitor.on('alert')            - عند تنبيه
monitor.on('metrics')          - عند تحديث مقاييس
